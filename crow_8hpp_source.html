72 // https://gist.github.com/fmela/591333
73 // This function produces a stack backtrace with demangled function & method names.
74 json get_backtrace(int skip = 1)
75 {
76  json result = json::array();
77 
78 #if defined(NLOHMANN_CROW_HAVE_DLFCN_H) && defined(NLOHMANN_CROW_HAVE_EXECINFO_H)
79  void* callstack[128];
80  const int nMaxFrames = sizeof(callstack) / sizeof(callstack[0]);
81  char buf[1024];
82  int nFrames = backtrace(callstack, nMaxFrames);
83  char** symbols = backtrace_symbols(callstack, nFrames);
84 
85  for (int i = skip; i < nFrames; i++)
86  {
87  Dl_info info;
88  if (dladdr(callstack[i], &info) && info.dli_sname)
89  {
90  char* demangled = nullptr;
91  int status = -1;
92  if (info.dli_sname[0] == '_')
93  {
94 #ifdef NLOHMANN_CROW_HAVE_CXXABI_H
95  demangled = abi::__cxa_demangle(info.dli_sname, nullptr, nullptr, &status);
96 #endif
97  }
98 
99  const std::string function_name = (status == 0 ? demangled : info.dli_sname == nullptr ? symbols[i] : info.dli_sname);
100 
101  snprintf(buf, sizeof(buf), "%-3d %*p %s + %zd\n",
102  i, int(2 + sizeof(void*) * 2), callstack[i],
103  function_name.c_str(),
104  (char*) callstack[i] - (char*) info.dli_saddr);
105  //std::cout << buf << std::endl;
106 
107  json entry;
108  entry["function"] = function_name;
109 
110  if (not function_name.empty())
111  {
112  if (function_name.substr(0, 5) == "std::" or function_name.substr(0, 2) == "__")
113  {
114  entry["in_app"] = false;
115  }
116  }
117 
118  result.push_back(entry);
119 
120  free(demangled);
121  }
122  else
123  {
124  snprintf(buf, sizeof(buf), "%-3d %*p %s\n",
125  i, int(2 + sizeof(void*) * 2), callstack[i], symbols[i]);
126  }
127  }
128  free(symbols);
129 #endif
130 
131  return result;
132 }

183 std::string generate_uuid()
184 {
185  std::random_device random_device;
186  std::default_random_engine random_engine(random_device());
187  std::uniform_int_distribution<int> uniform_dist(0, 15);
188 
189  std::string result(32, ' ');
190 
191  for (std::size_t i = 0; i < result.size(); ++i)
192  {
193  // the 12th character must be a '4'
194  if (i == 12)
195  {
196  result[i] = '4';
197  }
198  else
199  {
200  // get a random number from 0..15
201  const auto r = static_cast<char>(uniform_dist(random_engine));
202  if (r < 10)
203  {
204  result[i] = '0' + r;
205  }
206  else
207  {
208  result[i] = 'a' + r - static_cast<char>(10);
209  }
210  }
211  }
212 
213  return result;
214 }
