71 // https://gist.github.com/fmela/591333
72 // This function produces a stack backtrace with demangled function & method names.
73 json get_backtrace(int skip = 1)
74 {
75  json result = json::array();
76 
77 #if defined(NLOHMANN_CROW_HAVE_DLFCN_H) && defined(NLOHMANN_CROW_HAVE_EXECINFO_H)
78  void* callstack[128];
79  const int nMaxFrames = sizeof(callstack) / sizeof(callstack[0]);
80  char buf[1024];
81  int nFrames = backtrace(callstack, nMaxFrames);
82  char** symbols = backtrace_symbols(callstack, nFrames);
83 
84  for (int i = skip; i < nFrames; i++)
85  {
86  Dl_info info;
87  if (dladdr(callstack[i], &info) && info.dli_sname)
88  {
89  char* demangled = nullptr;
90  int status = -1;
91  if (info.dli_sname[0] == '_')
92  {
93 #ifdef NLOHMANN_CROW_HAVE_CXXABI_H
94  demangled = abi::__cxa_demangle(info.dli_sname, nullptr, nullptr, &status);
95 #endif
96  }
97 
98  const std::string function_name = (status == 0 ? demangled : info.dli_sname == nullptr ? symbols[i] : info.dli_sname);
99 
100  snprintf(buf, sizeof(buf), "%-3d %*p %s + %zd\n",
101  i, int(2 + sizeof(void*) * 2), callstack[i],
102  function_name.c_str(),
103  (char*) callstack[i] - (char*) info.dli_saddr);
104  //std::cout << buf << std::endl;
105 
106  json entry;
107  entry["function"] = function_name;
108 
109  if (not function_name.empty())
110  {
111  if (function_name.substr(0, 5) == "std::" or function_name.substr(0, 2) == "__")
112  {
113  entry["in_app"] = false;
114  }
115  }
116 
117  result.push_back(entry);
118 
119  free(demangled);
120  }
121  else
122  {
123  snprintf(buf, sizeof(buf), "%-3d %*p %s\n",
124  i, int(2 + sizeof(void*) * 2), callstack[i], symbols[i]);
125  }
126  }
127  free(symbols);
128 #endif
129 
130  return result;
131 }

181 std::string generate_uuid()
182 {
183  std::random_device random_device;
184  std::default_random_engine random_engine(random_device());
185  std::uniform_int_distribution<int> uniform_dist(0, 15);
186 
187  std::string result(32, ' ');
188 
189  for (size_t i = 0; i < 32; ++i)
190  {
191  if (i == 12)
192  {
193  result[i] = '4';
194  }
195  else
196  {
197  const auto r = uniform_dist(random_engine);
198  if (r < 10)
199  {
200  result[i] = '0' + r;
201  }
202  else
203  {
204  result[i] = 'a' + static_cast<char>(r - 10);
205  }
206  }
207  }
208 
209  return result;
210 }
